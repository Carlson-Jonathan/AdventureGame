    if(myfile.is_open()) {

            cout << "Full line: " << tempLine << endl;

            if (found!=std::string::npos && !detected) {
                firstWord = tempLine.substr(0, found);
                if(firstWord == character) {
                    cout << "You found a dragon!" << endl;
                    detected = true;
                }
            }
            if(tempLine == "" && detected)
                break;
            if(detected) {
                cout << "Adding a line... ";
                spriteData.rawData[lines] = tempLine;
